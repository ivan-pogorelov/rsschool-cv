*My task is to rotate a matrix 90 degree to the left.*

```javascript
let rotateMatrix = (arr) => {
    let result = [];

    for (let i = arr[0].length - 1; i >= 0; i--) {
        let temp = [];
        for (let j = 0; j < arr.length; j++) {
            temp.push(arr[j][i]);
        }
        result.push(temp);
    }
  
    return result;
}
console.log(rotateMatrix([[-1,4,5],[2,3,4]]))
```
